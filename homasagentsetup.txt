python"""
Tool Orchestration for Claude Code
Manages tool selection and execution within Claude Code environment
"""

import json
from typing import Dict, List, Any, Optional
import asyncio

class ToolOrchestrator:
    """Intelligent tool selection and execution"""
    
    def __init__(self, config_path: str = "./config/tools_config.yaml"):
        self.config = self._load_config(config_path)
        self.tool_history = []
        self.tool_metrics = {}
    
    async def select_tool(self, research_need: str, context: Dict) -> str:
        """
        Select optimal tool based on research need and context
        
        This will be executed by Claude Code with access to all MCP tools
        """
        
        # Tool selection logic based on the matrix
        tool_matrix = self.config['tool_selection_matrix']['research_needs']
        
        if research_need in tool_matrix:
            primary = tool_matrix[research_need]['primary']
            fallback = tool_matrix[research_need].get('fallback', None)
            
            # Return tool selection for Claude Code to execute
            return {
                'primary': primary,
                'fallback': fallback,
                'reason': f"Selected based on research need: {research_need}"
            }
        
        return None
    
    async def execute_tool_cascade(self, tools: List[str], query: Any) -> Dict:
        """
        Execute tool cascade with fallback logic
        Claude Code will handle actual tool execution
        """
        
        cascade_prompt = f"""
        Execute the following tool cascade:
        Primary Tool: {tools[0]}
        Query/Parameters: {json.dumps(query)}
        
        If the primary tool fails or returns insufficient results:
        - Try fallback tools in order: {tools[1:]}
        - Document each attempt
        - Return the first successful result
        
        Tool execution protocol:
        1. Log the tool call
        2. Execute with appropriate parameters
        3. Validate the response
        4. If failed, proceed to next tool
        5. Return consolidated results
        """
        
        return {'prompt': cascade_prompt, 'tools': tools, 'query': query}

        for path in [self.sessions_path / "active", 
                     self.sessions_path / "archived",
                     self.context_path / "entities"]:
            path.mkdir(parents=True, exist_ok=True)
        
        # Load session index
        self.index_path = self.sessions_path / "index.json"
        self.sessions = self._load_session_index()
        
        # Load knowledge base
        self.knowledge_base = self._load_knowledge_base()
        
    def _load_session_index(self) -> Dict[str, ResearchSession]:
        """Load or create session index"""
        if self.index_path.exists():
            with open(self.index_path, 'r') as f:
                data = json.load(f)
                return {k: ResearchSession(**v) for k, v in data.items()}
        return {}
    
    def _save_session_index(self):
        """Save session index"""
        data = {k: asdict(v) for k, v in self.sessions.items()}
        with open(self.index_path, 'w') as f:
            json.dump(data, f, indent=2)
    
    def _load_knowledge_base(self) -> Dict[str, Any]:
        """Load all knowledge base files"""
        kb = {}
        for yaml_file in self.kb_path.glob("*.yaml"):
            with open(yaml_file, 'r') as f:
                kb[yaml_file.stem] = yaml.safe_load(f)
        return kb
    
    def create_session(self, 
                      entity_name: str, 
                      entity_type: str,
                      objective: str,
                      parent_session: Optional[str] = None) -> ResearchSession:
        """Create new research session"""
        
        # Generate entity ID
        entity_id = self._get_or_create_entity_id(entity_name, entity_type)
        
        # Check for existing active session
        existing = self.find_active_session(entity_id)
        if existing:
            print(f"Found existing session for {entity_name}: {existing.session_id}")
            return self.resume_session(existing.session_id)
        
        # Create new session
        timestamp = datetime.now()
        session_id = f"{entity_id}_{timestamp.strftime('%Y%m%d_%H%M%S')}"
        
        session = ResearchSession(
            session_id=session_id,
            entity_id=entity_id,
            entity_name=entity_name,
            entity_type=entity_type,
            objective=objective,
            status="active",
            created_at=timestamp.isoformat(),
            updated_at=timestamp.isoformat(),
            parent_session=parent_session
        )
        
        # Create session directory
        session_dir = self.sessions_path / "active" / session_id
        session_dir.mkdir(parents=True, exist_ok=True)
        
        # Initialize session files
        self._initialize_session_files(session_dir, session)
        
        # Update parent session if exists
        if parent_session and parent_session in self.sessions:
            self.sessions[parent_session].child_sessions.append(session_id)
        
        # Save to index
        self.sessions[session_id] = session
        self._save_session_index()
        
        print(f"Created new session: {session_id}")
        return session
    
    def _get_or_create_entity_id(self, entity_name: str, entity_type: str) -> str:
        """Get or create normalized entity ID"""
        # Normalize entity name
        normalized = entity_name.lower().replace(" ", "_").replace(",", "")
        entity_id = f"{entity_type}_{normalized}"
        
        # Create entity context store if doesn't exist
        entity_path = self.context_path / "entities" / entity_id
        if not entity_path.exists():
            entity_path.mkdir(parents=True, exist_ok=True)
            
            # Initialize entity profile
            profile = {
                "entity_id": entity_id,
                "entity_name": entity_name,
                "entity_type": entity_type,
                "created_at": datetime.now().isoformat(),
                "research_sessions": [],
                "key_findings": {},
                "relationships": []
            }
            
            with open(entity_path / "profile.json", 'w') as f:
                json.dump(profile, f, indent=2)
        
        return entity_id
    
    def _initialize_session_files(self, session_dir: Path, session: ResearchSession):
        """Initialize session directory with required files"""
        
        # Session metadata
        with open(session_dir / "session.json", 'w') as f:
            json.dump(asdict(session), f, indent=2)
        
        # Research context
        context = {
            "session_id": session.session_id,
            "entity": {
                "id": session.entity_id,
                "name": session.entity_name,
                "type": session.entity_type
            },
            "objective": session.objective,
            "knowledge_base": list(self.knowledge_base.keys()),
            "agent_outputs": {},
            "tool_calls": [],
            "findings": {},
            "checkpoints": {}
        }
        
        with open(session_dir / "context.json", 'w') as f:
            json.dump(context, f, indent=2)
        
        # Create subdirectories
        for subdir in ["checkpoints", "outputs", "evidence", "tool_logs"]:
            (session_dir / subdir).mkdir(exist_ok=True)
    
    def find_active_session(self, entity_id: str) -> Optional[ResearchSession]:
        """Find active session for entity"""
        for session in self.sessions.values():
            if session.entity_id == entity_id and session.status == "active":
                return session
        return None
    
    def resume_session(self, session_id: str) -> ResearchSession:
        """Resume existing session with full context"""
        if session_id not in self.sessions:
            raise ValueError(f"Session {session_id} not found")
        
        session = self.sessions[session_id]
        session_dir = self.sessions_path / "active" / session_id
        
        # Load context
        with open(session_dir / "context.json", 'r') as f:
            context = json.load(f)
        
        # Load entity history
        entity_context = self._load_entity_context(session.entity_id)
        
        print(f"Resumed session: {session_id}")
        print(f"Entity: {session.entity_name}")
        print(f"Previous checkpoints: {session.checkpoint_count}")
        print(f"Related sessions: {len(entity_context.get('research_sessions', []))}")
        
        return session, context, entity_context
    
    def _load_entity_context(self, entity_id: str) -> Dict:
        """Load complete entity context including history"""
        entity_path = self.context_path / "entities" / entity_id
        
        context = {}
        
        # Load profile
        if (entity_path / "profile.json").exists():
            with open(entity_path / "profile.json", 'r') as f:
                context['profile'] = json.load(f)
        
        # Load research history
        if (entity_path / "research_history.json").exists():
            with open(entity_path / "research_history.json", 'r') as f:
                context['history'] = json.load(f)
        
        # Load cached findings
        if (entity_path / "cached_findings.json").exists():
            with open(entity_path / "cached_findings.json", 'r') as f:
                context['cached_findings'] = json.load(f)
        
        return context
    
    def switch_context(self, from_session: str, to_entity: str, 
                       objective: str = None) -> ResearchSession:
        """Switch research context to different entity"""
        
        # Pause current session
        if from_session in self.sessions:
            self.pause_session(from_session)
        
        # Check if target entity has active session
        entity_type = "company"  # Default, could be parameterized
        to_entity_id = self._get_or_create_entity_id(to_entity, entity_type)
        
        existing = self.find_active_session(to_entity_id)
        if existing:
            return self.resume_session(existing.session_id)
        
        # Create new session for target entity
        if not objective:
            objective = f"Continue research on {to_entity}"
        
        return self.create_session(
            entity_name=to_entity,
            entity_type=entity_type,
            objective=objective,
            parent_session=from_session
        )
    
    def pause_session(self, session_id: str):
        """Pause active session and save state"""
        if session_id not in self.sessions:
            return
        
        session = self.sessions[session_id]
        session.status = "paused"
        session.updated_at = datetime.now().isoformat()
        
        # Save checkpoint
        self.create_checkpoint(session_id, "pause")
        
        self._save_session_index()
        print(f"Paused session: {session_id}")
    
    def create_checkpoint(self, session_id: str, checkpoint_type: str = "auto"):
        """Create session checkpoint"""
        session = self.sessions[session_id]
        session_dir = self.sessions_path / "active" / session_id
        
        checkpoint_id = f"checkpoint_{session.checkpoint_count:03d}_{checkpoint_type}"
        checkpoint_dir = session_dir / "checkpoints" / checkpoint_id
        checkpoint_dir.mkdir(parents=True, exist_ok=True)
        
        # Load current context
        with open(session_dir / "context.json", 'r') as f:
            context = json.load(f)
        
        # Create checkpoint
        checkpoint = {
            "checkpoint_id": checkpoint_id,
            "session_id": session_id,
            "type": checkpoint_type,
            "timestamp": datetime.now().isoformat(),
            "context_hash": hashlib.sha256(
                json.dumps(context, sort_keys=True).encode()
            ).hexdigest(),
            "context": context
        }
        
        # Save checkpoint
        with open(checkpoint_dir / "checkpoint.json", 'w') as f:
            json.dump(checkpoint, f, indent=2)
        
        # Update session
        session.checkpoint_count += 1
        session.context_hash = checkpoint["context_hash"]
        self._save_session_index()
        
        print(f"Created checkpoint: {checkpoint_id}")
    
    def get_handoff_context(self, session_ids: List[str]) -> Dict:
        """Prepare handoff context for multiple sessions"""
        handoff = {
            "timestamp": datetime.now().isoformat(),
            "sessions": {},
            "entities": {},
            "relationships": []
        }
        
        for session_id in session_ids:
            if session_id not in self.sessions:
                continue
            
            session = self.sessions[session_id]
            session_dir = self.sessions_path / "active" / session_id
            
            # Load session context
            with open(session_dir / "context.json", 'r') as f:
                context = json.load(f)
            
            handoff["sessions"][session_id] = {
                "session": asdict(session),
                "context": context,
                "last_checkpoint": session.checkpoint_count
            }
            
            # Load entity context
            if session.entity_id not in handoff["entities"]:
                handoff["entities"][session.entity_id] = self._load_entity_context(
                    session.entity_id
                )
        
        # Load relationships
        rel_path = self.context_path / "relationships.json"
        if rel_path.exists():
            with open(rel_path, 'r') as f:
                handoff["relationships"] = json.load(f)
        
        return handoff
    
    def merge_findings(self, session_id: str, findings: Dict):
        """Merge new findings into entity knowledge"""
        session = self.sessions[session_id]
        entity_path = self.context_path / "entities" / session.entity_id
        
        # Load existing cached findings
        cache_path = entity_path / "cached_findings.json"
        if cache_path.exists():
            with open(cache_path, 'r') as f:
                cached = json.load(f)
        else:
            cached = {
                "entity_id": session.entity_id,
                "findings": {},
                "sources": {},
                "last_updated": {}
            }
        
        # Merge new findings
        for key, value in findings.items():
            if key not in cached["findings"]:
                cached["findings"][key] = []
            
            # Add with timestamp and session reference
            cached["findings"][key].append({
                "value": value,
                "session_id": session_id,
                "timestamp": datetime.now().isoformat()
            })
            
            cached["last_updated"][key] = datetime.now().isoformat()
        
        # Save updated cache
        with open(cache_path, 'w') as f:
            json.dump(cached, f, indent=2)
        
        print(f"Merged {len(findings)} findings for {session.entity_name}")

    def complete_session(self, session_id: str, quality_score: float):
        """Complete and archive session"""
        session = self.sessions[session_id]
        session.status = "completed"
        session.quality_score = quality_score
        session.updated_at = datetime.now().isoformat()
        
        # Move to archived
        active_dir = self.sessions_path / "active" / session_id
        archived_dir = self.sessions_path / "archived" / session_id
        
        if active_dir.exists():
            active_dir.rename(archived_dir)
        
        self._save_session_index()
        print(f"Completed session: {session_id} (Quality: {quality_score:.2f})")

import asyncio

class HOMASOrchestrator:
    """Enhanced orchestrator with multi-entity session management"""
    
    def __init__(self):
        self.session_manager = SessionManager()
        self.current_session = None
        self.entity_cache = {}
        
    async def research_entity(self, 
                             entity_name: str,
                             entity_type: str,
                             objective: str,
                             use_cache: bool = True) -> Dict:
        """
        Research an entity with session management
        
        Example:
            await orchestrator.research_entity("Apple Inc", "company", 
                                              "Analyze Q4 2024 performance")
        """
        
        # Check for cached findings if switching back
        entity_id = self.session_manager._get_or_create_entity_id(
            entity_name, entity_type
        )
        
        if use_cache and entity_id in self.entity_cache:
            print(f"Using cached context for {entity_name}")
            cached_context = self.entity_cache[entity_id]
        else:
            cached_context = None
        
        # Create or resume session
        session = self.session_manager.create_session(
            entity_name=entity_name,
            entity_type=entity_type,
            objective=objective,
            parent_session=self.current_session.session_id if self.current_session else None
        )
        
        self.current_session = session
        
        # Load knowledge base and entity context
        kb = self.session_manager.knowledge_base
        entity_context = self.session_manager._load_entity_context(entity_id)
        
        # Prepare research prompt with context
        research_prompt = self._prepare_research_prompt(
            session, objective, kb, entity_context, cached_context
        )
        
        # Execute research pipeline
        results = await self._execute_research_pipeline(
            session, research_prompt
        )
        
        # Cache findings
        self.entity_cache[entity_id] = results['key_findings']
        
        # Merge findings into entity knowledge
        self.session_manager.merge_findings(
            session.session_id, 
            results['key_findings']
        )
        
        return results
    
    async def research_sequence(self, sequence: List[Dict]) -> Dict:
        """
        Execute a sequence of research tasks with context switching
        
        Example:
            sequence = [
                {"entity": "Apple Inc", "type": "company", 
                 "objective": "Q4 performance"},
                {"entity": "Microsoft", "type": "company", 
                 "objective": "Cloud revenue"},
                {"entity": "Apple Inc", "type": "company", 
                 "objective": "Compare with previous findings"},
                {"entity": "Google", "type": "company", 
                 "objective": "AI initiatives"}
            ]
        """
        
        all_results = {}
        
        for idx, task in enumerate(sequence, 1):
            print(f"\n{'='*60}")
            print(f"Research Task {idx}/{len(sequence)}: {task['entity']}")
            print(f"Objective: {task['objective']}")
            print(f"{'='*60}")
            
            # Check if returning to previous entity
            use_cache = any(
                prev['entity'] == task['entity'] 
                for prev in sequence[:idx-1]
            )
            
            if use_cache:
                print(f"Note: Returning to {task['entity']} - will use cached context")
            
            # Execute research
            results = await self.research_entity(
                entity_name=task['entity'],
                entity_type=task.get('type', 'company'),
                objective=task['objective'],
                use_cache=use_cache
            )
            
            all_results[f"{task['entity']}_{idx}"] = results
            
            # Create checkpoint after each task
            self.session_manager.create_checkpoint(
                self.current_session.session_id,
                f"task_{idx}"
            )
        
        # Generate comparative analysis if multiple entities
        if len(set(task['entity'] for task in sequence)) > 1:
            comparison = await self._generate_comparison(all_results)
            all_results['comparison'] = comparison
        
        return all_results
    
    def _prepare_research_prompt(self,

    generate_summary_report(results, output_dir / f"summary_{timestamp}.md")
    
    return results

def generate_summary_report(results: Dict, output_path: Path):
    """Generate markdown summary report"""
    
    report = """# HOMAS Research Sequence Summary

## Research Sessions Executed
"""
    
    for key, result in results.items():
        if key == 'comparison':
            continue
        
        report += f"""
### {result.get('entity', key)}
- **Objective**: {result.get('objective', 'N/A')}
- **Session ID**: {result.get('session_id', 'N/A')}

- **Key Findings**: {len(result.get('key_findings', {}))}
- **Tool Calls**: {len(result.get('tool_calls', []))}
"""
    
    if 'comparison' in results:
        report += """
## Comparative Analysis
"""
        report += str(results['comparison'])
    
    with open(output_path, 'w') as f:
        f.write(report)
    
    print(f"Summary report saved to: {output_path}")

if __name__ == "__main__":
    asyncio.run(run_research_sequence())

Context Preservation: Each entity maintains its own knowledge store that persists across sessions
Efficient Handoffs: When switching between entities, the system automatically:

Saves current state
Loads relevant context for new entity
Maintains parent-child relationships


Cache Intelligence: When returning to a previously researched entity:

System recognizes return visit
Loads cached findings
Builds upon previous research rather than starting fresh


Knowledge Base Integration: All KB files are loaded and available to every agent
Relationship Tracking: System tracks relationships between entities for comparative analysis
Checkpoint System: Automatic checkpointing ensures no research is lost

    """
    
    def __init__(self, registry_path: str = "./context_store/entity_registry.json"):
        self.registry_path = Path(registry_path)
        self.registry = self._load_registry()
        self.verification_threshold = 0.85
        
    def _load_registry(self) -> Dict:
        """Load entity registry"""
        if self.registry_path.exists():
            with open(self.registry_path, 'r') as f:
                return json.load(f)
        
        return {
            "entities": {},
            "aliases": {},
            "relationships": {}
        }
    
    def _save_registry(self):
        """Save entity registry"""
        self.registry_path.parent.mkdir(parents=True, exist_ok=True)
        with open(self.registry_path, 'w') as f:
            json.dump(self.registry, f, indent=2)
    
    async def verify_entity(self, 
                           query_entity: str,
                           entity_type: str,
                           context: Dict = None) -> Dict:
        """
        Verify entity identification with confirmation
        
        Returns:
            {
                "verified": bool,
                "entity_id": str,
                "canonical_name": str,
                "confidence": float,
                "requires_confirmation": bool,
                "alternatives": list,
                "verification_method": str
            }
        """
        
        # Normalize query
        normalized = self._normalize_entity_name(query_entity)
        
        # Check exact match
        if normalized in self.registry['entities']:
            return {
                "verified": True,
                "entity_id": self.registry['entities'][normalized]['entity_id'],
                "canonical_name": self.registry['entities'][normalized]['canonical_name'],
                "confidence": 1.0,
                "requires_confirmation": False,
                "alternatives": [],
                "verification_method": "exact_match"
            }
        
        # Check aliases
        if normalized in self.registry['aliases']:
            entity_id = self.registry['aliases'][normalized]
            return {
                "verified": True,
                "entity_id": entity_id,
                "canonical_name": self.registry['entities'][entity_id]['canonical_name'],
                "confidence": 0.95,
                "requires_confirmation": False,
                "alternatives": [],
                "verification_method": "alias_match"
            }
        
        # Fuzzy matching
        matches = self._fuzzy_match(normalized)
        
        if matches and matches[0]['score'] > self.verification_threshold:
            # High confidence match
            return {
                "verified": True,
                "entity_id": matches[0]['entity_id'],
                "canonical_name": matches[0]['canonical_name'],
                "confidence": matches[0]['score'],
                "requires_confirmation": matches[0]['score'] < 0.95,
                "alternatives": matches[1:3],  # Top 3 alternatives
                "verification_method": "fuzzy_match"
            }
        
        # Ambiguous or new entity
        if matches:
            return {
                "verified": False,
                "entity_id": None,
                "canonical_name": query_entity,
                "confidence": matches[0]['score'] if matches else 0.0,
                "requires_confirmation": True,
                "alternatives": matches[:5],
                "verification_method": "ambiguous"
            }
        
        # Completely new entity
        return await self._handle_new_entity(query_entity, entity_type)
    
    def _normalize_entity_name(self, name: str) -> str:
        """Normalize entity name for matching"""
        
        # Convert to lowercase
        normalized = name.lower().strip()
        
        # Remove common suffixes
        suffixes = [
            ' inc.', ' inc', ' incorporated',
            ' corp.', ' corp', ' corporation',
            ' llc', ' ltd.', ' ltd', ' limited',
            ' co.', ' co', ' company',
            ' group', ' holdings', ' plc'
        ]
        
        for suffix in suffixes:
            if normalized.endswith(suffix):
                normalized = normalized[:-len(suffix)].strip()
        
        # Remove special characters
        normalized = ''.join(c for c in normalized if c.isalnum() or c.isspace())
        
        # Collapse multiple spaces
        normalized = ' '.join(normalized.split())
        
        return normalized
    
    def _fuzzy_match(self, query: str) -> List[Dict]:
        """Perform fuzzy matching against registry"""
        
        matches = []
        
        for entity_key, entity_data in self.registry['entities'].items():
            # Calculate similarity scores
            name_score = self._calculate_similarity(
                query, 
                entity_data['canonical_name'].lower()
            )
            
            # Check against aliases
            alias_scores = []
            for alias in entity_data.get('aliases', []):
                alias_scores.append(
                    self._calculate_similarity(query, alias.lower())
                )
            
            best_score = max([name_score] + alias_scores)
            
            if best_score > 0.6:  # Minimum threshold
                matches.append({
                    "entity_id": entity_data['entity_id'],
                    "canonical_name": entity_data['canonical_name'],
                    "score": best_score,
                    "match_type": "name" if best_score == name_score else "alias"
                })
        
        # Sort by score
        matches.sort(key=lambda x: x['score'], reverse=True)
        
        return matches
    
    def _calculate_similarity(self, str1: str, str2: str) -> float:
        """Calculate string similarity using multiple methods"""
        
        # Levenshtein distance
        lev_score = 1 - (Levenshtein.distance(str1, str2) / max(len(str1), len(str2)))
        
        # Jaccard similarity (word-level)
        words1 = set(str1.split())
        words2 = set(str2.split())
        
        if words1 or words2:
            jaccard = len(words1 & words2) / len(words1 | words2)
        else:
            jaccard = 0
        
        # Weighted average
        return (lev_score * 0.7) + (jaccard * 0.3)
    
    async def _handle_new_entity(self, entity_name: str, entity_type: str) -> Dict:
        """Handle completely new entity"""

        entity_id = self._generate_entity_id(entity_name, entity_type)

            "alternatives": [],

            "action_required": "confirm_new_entity"
        }

        hash_suffix = hashlib.sha256(
            f"{name}_{entity_type}".encode()
        ).hexdigest()[:8]
        
        return f"{base_id}_{hash_suffix}"

                    content=self._format_future_research(report_data),
                    confidence=1.0,
                    evidence_count=0
                )
            ]
        ))
        
        return sections
    
    def _generate_markdown(self, sections: List[ReportSection], data: Dict) -> str:
        """Generate Markdown report"""
        
        output_path = Path(f"./outputs/reports/{data['metadata']['session_id']}")
        output_path.mkdir(parents=True, exist_ok=True)
        
        md_content = f"""# {data['metadata']['title']}

## {data['metadata']['subtitle']}

**Date:** {data['metadata']['date']}  
**Session ID:** {data['metadata']['session_id']}  
**Overall Confidence:** {data['metadata']['confidence_level']:.1%}  
**Quality Score:** {data['metadata']['quality_score']:.2f}/1.00

---

"""
        
        # Add table of contents
        md_content += "## Table of Contents\n\n"
        for idx, section in enumerate(sections, 1):
            md_content += f"{idx}. [{section.title}](#{section.title.lower().replace(' ', '-')})\n"
            for sub_idx, subsection in enumerate(section.subsections, 1):
                md_content += f"   {idx}.{sub_idx}. [{subsection.title}](#{subsection.title.lower().replace(' ', '-')})\n"
        
        md_content += "\n---\n\n"
        
        # Add sections
        for section in sections:
            md_content += self._format_markdown_section(section)
        
        # Add appendices
        md_content += "\n\n## Appendices\n\n"
        for appendix in data['appendices']:
            md_content += f"### {appendix['title']}\n\n{appendix['content']}\n\n"
        
        # Save file
        md_path = output_path / f"report_{data['metadata']['session_id']}.md"
        with open(md_path, 'w') as f:
            f.write(md_content)
        
        return str(md_path)
    
    def _format_markdown_section(self, section: ReportSection, level: int = 2) -> str:
        """Format section as Markdown"""
        
        md = f"\n{'#' * level} {section.title}\n\n"
        
        # Add confidence badge
        confidence_badge = self._get_confidence_badge(section.confidence)
        md += f"*Confidence: {confidence_badge} | Evidence Points: {section.evidence_count}*\n\n"
        
        md += f"{section.content}\n\n"
        
        # Add charts
        for chart in section.charts:
            md += f"![{chart['title']}]({chart['path']})\n"
            md += f"*Figure: {chart['caption']}*\n\n"
        
        # Add tables
        for table in section.tables:
            md += self._format_markdown_table(table) + "\n\n"
        
        # Add subsections
        for subsection in section.subsections:
            md += self._format_markdown_section(subsection, level + 1)
        
        return md
    
    def _get_confidence_badge(self, confidence: float) -> str:
        """Get confidence level badge"""
        if confidence >= 0.9:
            return "🟢 **High** (≥90%)"
        elif confidence >= 0.7:
            return "🟡 **Medium** (70-89%)"
        else:
            return "🔴 **Low** (<70%)"
    
    def _format_markdown_table(self, table: Dict) -> str:
        """Format table in Markdown"""
        
        md = f"**{table['title']}**\n\n"
        
        # Headers
        headers = table['headers']
        md += "| " + " | ".join(headers) + " |\n"
        md += "| " + " | ".join(["---"] * len(headers)) + " |\n"
        
        # Rows
        for row in table['rows']:
            md += "| " + " | ".join(str(cell) for cell in row) + " |\n"
        
        return md
    
    def _generate_html(self, sections: List[ReportSection], data: Dict) -> str:
        """Generate HTML report using templates"""
        
        template = self.jinja_env.get_template("mckinsey_report.html")
        
        html_content = template.render(
            metadata=data['metadata'],
            sections=sections,
            appendices=data['appendices'],
            timestamp=datetime.now().isoformat()
        )
        
        output_path = Path(f"./outputs/reports/{data['metadata']['session_id']}")
        html_path = output_path / f"report_{data['metadata']['session_id']}.html"
        
        with open(html_path, 'w') as f:
            f.write(html_content)
        
        return str(html_path)
    
    def _generate_pdf(self, sections: List[ReportSection], data: Dict) -> str:
        """Generate PDF report from HTML"""
        
        # First generate HTML
        html_path = self._generate_html(sections, data)
        
        # Convert to PDF
        output_path = Path(f"./outputs/reports/{data['metadata']['session_id']}")
        pdf_path = output_path / f"report_{data['metadata']['session_id']}.pdf"
        
        # Load CSS for styling
        css = CSS(string=self._get_pdf_css())
        
        # Generate PDF
        HTML(filename=html_path).write_pdf(pdf_path, stylesheets=[css])
        
        return str(pdf_path)

    # Helper methods for data extraction and formatting
    
    def _calculate_overall_confidence(self, results: Dict) -> float:
        """Calculate overall confidence level"""
        confidences = []
        
        for agent_output in results.get('agent_outputs', {}).values():
            if 'confidence' in agent_output:
                confidences.append(agent_output['confidence'])
        
        return sum(confidences) / len(confidences) if confidences else 0.0
    
    def _generate_executive_summary(self, results: Dict) -> Dict:
        """Generate executive summary section"""
        
        # Extract BLUF from synthesis engine output
        synthesis = results.get('agent_outputs', {}).get('synthesis_engine_executive_output', {})
        
        return {
            "content": synthesis.get('executive_summary', 'No summary available'),
            "bluf": synthesis.get('bluf', 'Key findings pending'),
            "confidence": synthesis.get('confidence', 0.0),
            "evidence_count": len(results.get('tool_calls', []))
        }
    
    def _extract_key_findings(self, results: Dict) -> List[Dict]:
        """Extract and structure key findings"""
        
        findings = []
        
        for agent_name, output in results.get('agent_outputs', {}).items():
            if 'findings' in output:
                for finding in output['findings']:
                    findings.append({
                        "title": finding.get('title', 'Finding'),
                        "summary": finding.get('summary', ''),
                        "confidence": finding.get('confidence', 0.0),
                        "source": agent_name,
                        "evidence": finding.get('evidence', [])
                    })
        
        # Sort by confidence
        findings.sort(key=lambda x: x['confidence'], reverse=True)
        
        return findings[:10]  # Top 10 findings
    
    def _document_methodology(self, results: Dict) -> Dict:
        """Document research methodology"""
        
        return {
            "description": "Multi-agent research using HOMAS framework",
            "agents_used": list(results.get('agent_outputs', {}).keys()),
            "tools_deployed": self._extract_tools_used(results),
            "sources": self._extract_sources(results),
            "quality_controls": [
                "CRAAP test validation",
                "Multi-tool verification",
                "Adversarial validation",
                "LLM-as-Judge scoring"
            ]
        }
    
    def _extract_tools_used(self, results: Dict) -> List[str]:
        """Extract unique tools used"""
        tools = set()
        
        for call in results.get('tool_calls', []):
            tools.add(call.get('tool', 'unknown'))
        
        return sorted(list(tools))
    
    def _extract_sources(self, results: Dict) -> List[Dict]:
        """Extract and categorize sources"""
        sources = []
        
        for call in results.get('tool_calls', []):
            if 'result' in call and 'source' in call['result']:
                sources.append({
                    "url": call['result']['source'],
                    "type": call.get('tool', ''),
                    "timestamp": call.get('timestamp', '')
                })
        
        return sources
    
    def _generate_recommendations(self, results: Dict) -> List[Dict]:
        """Generate strategic recommendations"""
        
        synthesis = results.get('agent_outputs', {}).get('synthesis_engine_executive_output', {})
        recs = synthesis.get('recommendations', [])
        
        formatted_recs = []
        for rec in recs:
            formatted_recs.append({
                "action": rec.get('action', ''),
                "rationale": rec.get('rationale', ''),
                "priority": rec.get('priority', 'medium'),
                "timeline": rec.get('timeline', 'immediate'),
                "resources": rec.get('resources', [])
            })
        
        return formatted_recs
    
    def _identify_limitations(self, results: Dict) -> List[str]:
        """Identify research limitations"""
        
        limitations = []
        
        # Check for handoff items
        if results.get('handoff_packages'):
            limitations.append("Some areas require human expertise for complete analysis")
        
        # Check confidence scores
        low_confidence = [
            agent for agent, output in results.get('agent_outputs', {}).items()
            if output.get('confidence', 1.0) < 0.7
        ]
        
        if low_confidence:
            limitations.append(f"Limited confidence in: {', '.join(low_confidence)}")
        
        # Check tool failures
        tool_failures = [
            call for call in results.get('tool_calls', [])
            if call.get('status') == 'failed'
        ]
        
        if tool_failures:
            limitations.append(f"Tool access limitations: {len(tool_failures)} failures")
        
        return limitations
    
    def _identify_future_research(self, results: Dict) -> List[Dict]:
        """Identify future research opportunities"""
        
        opportunities = []
        
        # From validation orchestrator
        validation = results.get('agent_outputs', {}).get('validation_orchestrator_eitl_handoff', {})
        
        for handoff in validation.get('handoff_packages', []):
            opportunities.append({
                "area": handoff.get('area', ''),
                "rationale": handoff.get('reason', ''),
                "expertise_required": handoff.get('expertise', '')
            })
        
        return opportunities
    
    def _prepare_appendices(self, results: Dict) -> List[Dict]:
        """Prepare report appendices"""
        
        appendices = []
        
        # Tool call log
        appendices.append({
            "title": "Appendix A: Tool Utilization Log",
            "content": self._format_tool_log(results.get('tool_calls', []))
        })
        
        # Source bibliography
        appendices.append({
            "title": "Appendix B: Source Bibliography",
            "content": self._format_bibliography(results)
        })
        
        # Confidence scoring details
        appendices.append({
            "title": "Appendix C: Confidence Scoring Methodology",
            "content": self._format_confidence_methodology()
        })
        
        # Data tables
        appendices.append({
            "title": "Appendix D: Supporting Data Tables",
            "content": self._format_data_tables(results)
        })
        
        return appendices
    
    def _format_tool_log(self, tool_calls: List[Dict]) -> str:
        """Format tool utilization log"""
        
        log = "| Timestamp | Tool | Query/URL | Status | Result Summary |\n"
        log += "|-----------|------|-----------|--------|----------------|\n"
        
        for call in tool_calls[:50]:  # Limit to 50 for brevity
            log += f"| {call.get('timestamp', '')[:19]} "
            log += f"| {call.get('tool', '')} "
            log += f"| {call.get('query', '')[:30]}... "
            log += f"| {call.get('status', '')} "
            log += f"| {call.get('result_summary', '')[:50]}... |\n"
        
        return log
    
    def _format_bibliography(self, results: Dict) -> str:
        """Format source bibliography"""
        
        sources = self._extract_sources(results)
        
        bib = ""
        for idx, source in enumerate(sources, 1):
            bib += f"{idx}. {source['url']} (Accessed via {source['type']} on {source['timestamp'][:10]})\n"
        
        return bib
    
    def _format_confidence_methodology(self) -> str:
        """Explain confidence scoring methodology"""
        
        return """
Confidence scores are calculated based on:

1. **Source Quality (40%)**
   - Authority and credibility of sources
   - Recency and relevance of information
   - Cross-verification across multiple sources

2. **Tool Agreement (30%)**
   - Consistency across different tools
   - Convergence of findings
   - Absence of contradictions

3. **Evidence Volume (20%)**
   - Number of supporting data points
   - Breadth of source coverage
   - Depth of analysis

4. **Validation Results (10%)**
   - Adversarial validation outcomes
   - Quality scoring from LLM-as-Judge
   - Identified limitations and gaps

Scores are normalized to a 0-100% scale with the following interpretations:
- High (≥90%): Strong confidence, multiple verified sources
- Medium (70-89%): Reasonable confidence, some verification
- Low (<70%): Limited confidence, requires additional research
"""

    def _format_key_takeaways(self, data: Dict) -> str:
        """Format key takeaways"""
        takeaways = []
        
        for finding in data['key_findings'][:5]:
            takeaways.append(f"• {finding['title']}: {finding['summary']}")
        
        return "\n".join(takeaways)
    
    def _format_critical_insights(self, data: Dict) -> str:
        """Format critical insights"""
        
        insights = data.get('executive_summary', {}).get('critical_insights', [])
        
        if not insights:
            return "No critical insights identified requiring immediate attention."
        
        formatted = []
        for insight in insights:
            formatted.append(f"**{insight.get('title', '')}**\n{insight.get('description', '')}")
        
        return "\n\n".join(formatted)
    
    def _generate_finding_sections(self, data: Dict) -> List[ReportSection]:
        """Generate detailed finding sections"""
        
        sections = []
        
        for finding in data['key_findings']:
            sections.append(ReportSection(
                title=finding['title'],
                content=finding['summary'],
                confidence=finding['confidence'],
                evidence_count=len(finding.get('evidence', [])),
                tables=[self._create_evidence_table(finding['evidence'])] if finding.get('evidence') else []
            ))
        
        return sections
    
    def _create_evidence_table(self, evidence: List[Dict]) -> Dict:
        """Create evidence table"""
        
        return {
            "title": "Supporting Evidence",
            "headers": ["Source", "Finding", "Confidence"],
            "rows": [
                [e.get('source', ''), e.get('finding', ''), f"{e.get('confidence', 0):.0%}"]
                for e in evidence
            ]
        }
    
    def _generate_recommendation_sections(self, data: Dict) -> List[ReportSection]:
        """Generate recommendation sections"""
        
        sections = []
        
        for rec in data['recommendations']:
            sections.append(ReportSection(
                title=rec['action'],
                content=f"{rec['rationale']}\n\n**Timeline:** {rec['timeline']}\n**Priority:** {rec['priority']}",
                confidence=0.85,
                evidence_count=5
            ))
        
        return sections
    
    def _format_data_sources(self, data: Dict) -> str:
        """Format data sources section"""
        
        sources_by_type = {}
        for source in data['methodology']['sources']:
            source_type = source['type']
            if source_type not in sources_by_type:
                sources_by_type[source_type] = []
            sources_by_type[source_type].append(source['url'])
        
        formatted = ""
        for source_type, urls in sources_by_type.items():
            formatted += f"**{source_type}** ({len(urls)} sources)\n"
            for url in urls[:3]:  # Show first 3
                formatted += f"- {url}\n"
            if len(urls) > 3:
                formatted += f"- ... and {len(urls)-3} more\n"
            formatted += "\n"
        
        return formatted
    
    def _format_framework(self, data: Dict) -> str:
        """Format analytical framework section"""
        
        return """
The HOMAS Framework employs a systematic approach:

1. **Objective Clarification** - Socratic questioning to refine research goals
2. **Hypothesis Generation** - Cross-disciplinary frameworks to generate testable hypotheses  
3. **Intelligence Gathering** - Multi-tool data extraction with validation
4. **Synthesis** - Pattern recognition and insight generation
5. **Validation** - Adversarial testing and quality scoring
6. **Recommendation Formation** - Evidence-based strategic guidance

Each phase employs specialized agents with domain expertise and tool access.
"""
    
    def _format_limitations(self, data: Dict) -> str:
        """Format limitations section"""
        
        limitations = data['limitations']
        
        if not limitations:
            return "No significant limitations identified in this research."
        
        formatted = "This research has the following limitations:\n\n"
        for limitation in limitations:
            formatted += f"• {limitation}\n"
        
        return formatted
    
    def _format_future_research(self, data: Dict) -> str:
        """Format future research section"""
        
        opportunities = data['future_research']
        
        if not opportunities:
            return "No specific future research needs identified."
        
        formatted = "The following areas warrant additional investigation:\n\n"
        
        for opp in opportunities:
            formatted += f"**{opp['area']}**\n"
            formatted += f"- Rationale: {opp['rationale']}\n"
            formatted += f"- Required expertise: {opp['expertise_required']}\n\n"
        
        return formatted

import json

    entity_name: str
    session_id: str
    timestamp: str
    query: str
    optimized_query: Optional[str]

    checksum: str

class ResearchAuditor:
    """
    Comprehensive audit system for research activities
    """
    
    def __init__(self, db_path: str = "./audit/research_audit.db"):
        self.db_path = Path(db_path)
        self.db_path.parent.mkdir(parents=True, exist_ok=True)
        self.conn = sqlite3.connect(str(self.db_path))
        self._initialize_database()
        
    def _initialize_database(self):
        """Initialize audit database schema"""
        
        cursor = self.conn.cursor()
        
        # Main audit table
        cursor.execute("""
            CREATE TABLE IF NOT EXISTS audit_records (
                audit_id TEXT PRIMARY KEY,
                entity_id TEXT NOT NULL,
                entity_name TEXT NOT NULL,
                session_id TEXT UNIQUE NOT NULL,
                timestamp DATETIME NOT NULL,
                query TEXT NOT NULL,
                optimized_query TEXT,
                research_type TEXT NOT NULL,
                quality_score REAL,
                confidence_level REAL,
                agent_sequence TEXT,
                tools_used TEXT,
                tool_call_count INTEGER,
                data_sources TEXT,
                key_findings TEXT,
                recommendations TEXT,
                limitations TEXT,
                handoffs TEXT,
                duration_minutes INTEGER,
                cost_estimate REAL,
                user TEXT,
                status TEXT,
                parent_session TEXT,
                child_sessions TEXT,
                report_paths TEXT,
                checksum TEXT,
                FOREIGN KEY (entity_id) REFERENCES entities(entity_id)
            )
        """)
        
        # Entity summary table
        cursor.execute("""
            CREATE TABLE IF NOT EXISTS entity_summaries (
                entity_id TEXT PRIMARY KEY,
                entity_name TEXT NOT NULL,
                entity_type TEXT,
                first_researched DATETIME,
                last_researched DATETIME,
                total_sessions INTEGER DEFAULT 0,
                total_findings INTEGER DEFAULT 0,
                average_quality REAL,
                average_confidence REAL,
                total_duration_hours REAL DEFAULT 0,
                total_cost REAL DEFAULT 0,
                top_topics TEXT,
                key_insights TEXT,
                current_status TEXT
            )
        """)
        
        # Findings evolution table
        cursor.execute("""
            CREATE TABLE IF NOT EXISTS findings_evolution (
                finding_id TEXT PRIMARY KEY,
                entity_id TEXT NOT NULL,
                finding_category TEXT,
                finding_text TEXT,
                first_discovered DATETIME,
                last_confirmed DATETIME,
                confidence_trend TEXT,
                supporting_sessions TEXT,
                contradicting_sessions TEXT,

                FOREIGN KEY (entity_id) REFERENCES entities(entity_id)
            )
        """)
        
        # Tool usage analytics table
        cursor.execute("""
            CREATE TABLE IF NOT EXISTS tool_analytics (

        # Create indexes for performance
        cursor.execute("CREATE INDEX IF NOT EXISTS idx_entity_id ON audit_records(entity_id)")
        cursor.execute("CREATE INDEX IF NOT EXISTS idx_timestamp ON audit_records(timestamp)")

        cursor.execute("CREATE INDEX IF NOT EXISTS idx_entity_findings ON findings_evolution(entity_id)")
        
        self.conn.commit()
    
    def record_research_session(self, 
                               session_data: Dict,
                               research_results: Dict,
                               report_paths: Dict) -> str:
        """
        Record a complete research session in the audit system
        """
        
        # Generate audit ID
        audit_id = self._generate_audit_id(session_data)
        
        # Calculate metrics
        metrics = self._calculate_metrics(research_results)
        
        # Create audit record
        record = AuditRecord(
            audit_id=audit_id,
            entity_id=session_data.get('entity_id'),
            entity_name=session_data.get('entity_name'),
            session_id=session_data.get('session_id'),
            timestamp=datetime.now().isoformat(),
            query=session_data.get('objective'),
            optimized_query=session_data.get('optimized_query'),
            research_type=metrics['research_type'],
            quality_score=research_results.get('quality_score', 0.0),
            confidence_level=metrics['avg_confidence'],
            agent_sequence=list(research_results.get('agent_outputs', {}).keys()),
            tools_used=metrics['unique_tools'],
            tool_call_count=metrics['tool_call_count'],
            data_sources=metrics['data_sources'],
            key_findings=research_results.get('key_findings', {}),
            recommendations=[r.get('action') for r in research_results.get('recommendations', [])],
            limitations=research_results.get('limitations', []),
            handoffs=[h.get('area') for h in research_results.get('handoff_packages', [])],
            duration_minutes=metrics['duration'],
            cost_estimate=metrics['cost'],
            user=session_data.get('user'),
            status=session_data.get('status', 'completed'),
            parent_session=session_data.get('parent_session'),
            child_sessions=session_data.get('child_sessions', []),
            report_paths=report_paths,
            checksum=self._generate_checksum(research_results)
        )
        
        # Store in database
        self._store_audit_record(record)
        
        # Update entity summary
        self._update_entity_summary(record, research_results)
        
        # Track findings evolution
        self._track_findings_evolution(record, research_results)
        
        # Record tool analytics
        self._record_tool_analytics(record, research_results)
        
        print(f"Audit recorded: {audit_id}")
        return audit_id
    
    def _generate_audit_id(self, session_data: Dict) -> str:
        """Generate unique audit ID"""
        
        components = [
            session_data.get('entity_id', ''),
            session_data.get('session_id', ''),
            datetime.now().isoformat()
        ]
        
        hash_input = '|'.join(components)
        return f"audit_{hashlib.sha256(hash_input.encode()).hexdigest()[:12]}"
    
    def _calculate_metrics(self, results: Dict) -> Dict:
        """Calculate session metrics"""
        
        metrics = {
            'research_type': self._determine_research_type(results),
            'avg_confidence': self._calculate_avg_confidence(results),
            'unique_tools': self._extract_unique_tools(results),
            'tool_call_count': len(results.get('tool_calls', [])),
            'data_sources': self._extract_data_sources(results),
            'duration': self._estimate_duration(results),
            'cost': self._estimate_cost(results)
        }
        
        return metrics
    
    def _determine_research_type(self, results: Dict) -> str:
        """Determine research depth type"""
        
        tool_count = len(results.get('tool_calls', []))
        agent_count = len(results.get('agent_outputs', {}))
        
        if tool_count < 10 and agent_count < 5:
            return "quick"
        elif tool_count < 30 and agent_count < 8:
            return "standard"
        elif tool_count < 60:
            return "deep"
        else:
            return "comprehensive"
    
    def _calculate_avg_confidence(self, results: Dict) -> float:
        """Calculate average confidence across agents"""
        
        confidences = []
        for output in results.get('agent_outputs', {}).values():
            if 'confidence' in output:
                confidences.append(output['confidence'])
        
        return sum(confidences) / len(confidences) if confidences else 0.0
    
    def _extract_unique_tools(self, results: Dict) -> List[str]:
        """Extract unique tools used"""
        
        tools = set()
        for call in results.get('tool_calls', []):
            tools.add(call.get('tool', 'unknown'))
        
        return sorted(list(tools))
    
    def _extract_data_sources(self, results: Dict) -> List[str]:
        """Extract data sources"""
        
        sources = []
        for call in results.get('tool_calls', []):
            if 'source' in call.get('result', {}):
                sources.append(call['result']['source'])
        
        return sources[:20]  # Limit to top 20
    
    def _estimate_duration(self, results: Dict) -> int:
        """Estimate research duration in minutes"""
        
        # Based on tool calls and complexity
        tool_count = len(results.get('tool_calls', []))
        
        # Rough estimation: 30 seconds per tool call + overhead
        return max(5, (tool_count * 0.5) + 10)
    
    def _estimate_cost(self, results: Dict) -> float:
        """Estimate research cost"""
        
        # Cost factors (example pricing)
        costs = {
            'brave_search': 0.001,
            'bright_data': 0.01,
            'parallel_ai': 0.05,
            'firecrawl': 0.005,
            'google_maps': 0.002,
            'claude_tokens': 0.00001  # per token
        }
        
        total_cost = 0.0
        
        for call in results.get('tool_calls', []):
            tool = call.get('tool', '')
            for key, cost in costs.items():
                if key in tool.lower():
                    total_cost += cost
                    break
        
        return round(total_cost, 4)
    
    def _generate_checksum(self, results: Dict) -> str:
        """Generate checksum for integrity verification"""
        
        # Create deterministic string from key results
        checksum_input = json.dumps({
            'findings': results.get('key_findings', {}),
            'quality': results.get('quality_score', 0),
            'agent_count': len(results.get('agent_outputs', {}))
        }, sort_keys=True)
        
        return hashlib.sha256(checksum_input.encode()).hexdigest()[:16]
    
    def _store_audit_record(self, record: AuditRecord):

        cursor = self.conn.cursor()

        # Insert record
        columns = ', '.join(record_dict.keys())
        placeholders = ', '.join(['?' for _ in record_dict])
        
        cursor.execute(
            f"INSERT OR REPLACE INTO audit_records ({columns}) VALUES ({placeholders})",

        )

    def _track_findings_evolution(self, record: AuditRecord, results: Dict):
        """Track how findings evolve over time"""
        
        cursor = self.conn.cursor()
        
        for category, findings in record.key_findings.items():
            if isinstance(findings, list):
                finding_list = findings
            elif isinstance(findings, dict):
                finding_list = [findings]
            else:
                finding_list = [str(findings)]
            
            for finding in finding_list:
                finding_text = str(finding)
                finding_id = hashlib.sha256(
                    f"{record.entity_id}_{category}_{finding_text[:100]}".encode()
                ).hexdigest()[:16]
                
                # Check if finding exists
                cursor.execute(
                    "SELECT * FROM findings_evolution WHERE finding_id = ?",
                    (finding_id,)
                )
                
                existing = cursor.fetchone()
                
                if existing:
                    # Update existing finding
                    cursor.execute("""
                        UPDATE findings_evolution
                        SET last_confirmed = ?,
                            supporting_sessions = supporting_sessions || ? || ',',
                            current_confidence = ?
                        WHERE finding_id = ?
                    """, (
                        record.timestamp,
                        record.session_id,
                        record.confidence_level,
                        finding_id
                    ))
                else:
                    # New finding
                    cursor.execute("""
                        INSERT INTO findings_evolution (
                            finding_id, entity_id, finding_category,
                            finding_text, first_discovered, last_confirmed,
                            supporting_sessions, current_confidence,
                            evolution_status
                        ) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)
                    """, (
                        finding_id,
                        record.entity_id,
                        category,
                        finding_text,
                        record.timestamp,
                        record.timestamp,
                        record.session_id,
                        record.confidence_level,
                        'new'
                    ))
        
        self.conn.commit()
    
    def _record_tool_analytics(self, record: AuditRecord, results: Dict):

                tool_stats[tool] = {
                    'count': 0,
                    'success': 0,
                    'total_time': 0,
                    'data_volume': 0
                }
            
            tool_stats[tool]['count'] += 1
            if call.get('status') == 'success':
                tool_stats[tool]['success'] += 1
            tool_stats[tool]['total_time'] += call.get('response_time_ms', 100)
            tool_stats[tool]['data_volume'] += call.get('data_kb', 1)
        
        # Store analytics
        for tool, stats in tool_stats.items():
            cursor.execute("""
                INSERT INTO tool_analytics (
                    session_id, tool_name, call_count,
                    success_rate, avg_response_time_ms,
                    data_volume_kb, timestamp
                ) VALUES (?, ?, ?, ?, ?, ?, ?)
            """, (
                record.session_id,
                tool,
                stats['count'],
                stats['success'] / stats['count'] if stats['count'] > 0 else 0,
                stats['total_time'] / stats['count'] if stats['count'] > 0 else 0,
                stats['data_volume'],
                record.timestamp
            ))
        
        self.conn.commit()

            GROUP BY research_type
            ORDER BY count DESC

            ORDER BY timestamp DESC

            report += f"(Quality: {row[3]:.2f}) [Session: {row[4]}]\n"

        report_path = Path("./audit/reports")
